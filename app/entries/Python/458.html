Je récupère les infos de cette manière :
  
Note : "lenght" vient du formulaire html "select".
   
# cat password_generator_project/password_generator_app/views.py  
from django.shortcuts import render  
from django.http import HttpResponse  
import random  
   
# Create your views here.  
   
def home(request):  
    return render(request, "password_generator_app_templates/home.html")  
   
def password_function(request):  
   
    characters = "abcdefghijklmnopqrstuvwxyz"  
    length = int(request.GET.get('lenght'))  
   
    the_password = ''  
    for x in range(length):  
            the_password += random.choice(characters)  
   
    return render(request, "password_generator_app_templates/password.html", {'password_dict_key': the_password})
   
 
  
Important ! On peut mettre une valeur par défaut :  
  
length = int(request.GET.get('lenght', 12))
